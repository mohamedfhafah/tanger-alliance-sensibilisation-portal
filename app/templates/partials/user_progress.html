{# Ce fichier contient des macros Jinja2 pour afficher la progression de l'utilisateur. Il utilise des barres de progression et des info-boxes pour visualiser les statistiques. NOTE: Les IDEs peuvent signaler des erreurs CSS/HTML dans ce fichier lorsque des variables Jinja2 sont utilisées dans les attributs style, mais ces erreurs sont généralement des faux positifs. #} {% macro render_progress_bar(title, value, max_value=100, color='primary', show_percentage=True) %}
{% if title %} {{ title }} {% if show_percentage %} {{ value }}/{{ max_value }} {% endif %} {% endif %}
{{ (0 if max_value == 0 else (value / max_value * 100))|round|int }}%
{% endmacro %} {% macro render_module_progress(modules, user_progress) %}

Progression des modules

{% set ns = namespace(completed_modules=0) %} {% for module in modules %} {% set progress = user_progress.get(module.id, None) %} {# Determine module status for clarity #} {% set current_module_is_completed = (progress and progress.completed == True) %} {% set current_module_score = (progress.score if progress and progress.score is not none else 0) %} {% if current_module_is_completed %} {% set ns.completed_modules = ns.completed_modules + 1 %} {% set color = 'success' %} {% elif progress and current_module_score > 0 %} {# Module started but not completed #} {% set color = 'primary' %} {% else %} {# Module not started or progress object missing #} {% set color = 'secondary' %} {% endif %} {# Determine progress value for the bar #} {% set progress_value = 100 if current_module_is_completed else current_module_score %} {{ render_progress_bar(module.title, progress_value, 100, color, False) }} {% endfor %}
{{ render_progress_bar('Progression totale', ns.completed_modules, modules|length, 'info') }}
{% endmacro %} {% macro render_user_stats(user, user_progress) %}
Modules terminés {{ user_progress|selectattr('completed', 'equalto', True)|list|length }}
Score moyen {% set scores = user_progress|selectattr('score', 'defined')|map(attribute='score')|list %} {% if scores|length > 0 %} {{ (scores|sum / scores|length)|round|int }}% {% else %} 0% {% endif %}
En cours {{ user_progress|selectattr('completed', 'equalto', False)|list|length }}
Badges {{ user.badges|default(0) }}
{% endmacro %}
