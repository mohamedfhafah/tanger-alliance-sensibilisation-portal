{# Ce fichier contient des macros Jinja2 qui génèrent des graphiques avec Chart.js. Structure importante à maintenir: 1. Les données sont calculées avec Jinja2 avant d'être insérées dans le JavaScript 2. Les variables JavaScript sont déclarées globalement (window.X) pour éviter les problèmes de scope 3. Le code JavaScript est externalisé dans dashboard-charts.js NOTE: Les IDEs peuvent signaler des erreurs JavaScript dans ce fichier mêlant Jinja2 et JS, mais ces erreurs sont généralement des faux positifs. Le code est fonctionnel lorsqu'il est rendu par Flask. #} {% macro render_module_completion_chart(user_progress, modules) %}

Progression par module

{# Préparation des données pour le graphique de progression par module #} {% set module_labels = [] %} {% set module_data = [] %} {% set completion_status = [] %} {% for module in modules %} {% do module_labels.append(module.title) %} {# Récupérer la progression de l'utilisateur pour ce module #} {% set progress = user_progress.get(module.id, None) %} {# Valeur du score pour le graphique (0-100) #} {% if progress and progress.score is not none %} {% do module_data.append(progress.score|round(1)) %} {% else %} {% do module_data.append(0) %} {% endif %} {# Statut de complétion (0 = non complété, 1 = complété) #} {% if progress and progress.completed %} {% do completion_status.append(1) %} {% else %} {% do completion_status.append(0) %} {% endif %} {% endfor %} {# Exportation des données en variables JavaScript #} {% endmacro %} {% macro render_overall_progress_chart(user_progress, modules) %}

Progression globale

{# Calcul des données pour le graphique de progression globale #} {% set total_modules = modules|length %} {% set ns_global = namespace(completed_modules=0) %} {# Compter les modules complétés à partir du dictionnaire user_progress #} {% for module in modules %} {% set progress = user_progress.get(module.id, None) %} {% if progress and progress.completed == True %} {% set ns_global.completed_modules = ns_global.completed_modules + 1 %} {% endif %} {% endfor %} {% set percentage_complete = ((ns_global.completed_modules / total_modules) * 100)|round|int if total_modules > 0 else 0 %} {# Exportation des données en variables JavaScript #} {% endmacro %} {% macro render_charts_script() %} {% endmacro %}
